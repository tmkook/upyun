又拍云图片上传jQuery插件

无需服务器中转，Form表单直传到又拍云。本插件依赖于 jQuery，请在 jQuery 后面引入 upyunFormUpload.js




Code:


$('#file').upyunForm({
  signUrl:"", //自定义签名地址，默认使用upyuntoken.php
  ready:function(up){
    $('#message').append("\r\n准备上传");
    //compress 是否开启压缩,width,height 压缩尺寸大小
    var param = {compress:1,width:200,height:200};
    up.upload(param); //开始上传
  },
  uploadBefore:function(obj){
    $('#message').append("\r\n上传中");
  },
  success:function(res){
    $('#message').append("\r\n上传成功:"+res.url);
  },
  error:function(res){
    $('#message').append("\r\n上传失败："+JSON.stringify(res));
  },
  sign:function(res){
    $('#message').append("Signature:"+res.signature);
    $('#message').append("\r\nPolicy:"+res.policy);
  }
});
